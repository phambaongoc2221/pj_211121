drop table `order`

create table `order`
(
    order_id   int auto_increment
        primary key,
    uID int references account(uID),
    ngayMua date,
    diachi nvarchar(100),
    priceTotal float default 0 null
);


drop table  CART
create table cart
(
    oderID     int auto_increment
        primary key,
    uID        int                      null,
    id         int                      null,
    DateCreate char(20)                 null,
    Quantity   int                      null,
    nguoiNhan  nvarchar(1000)  null,
    diaChiNhan nvarchar(1000)  null,
    soDT       char(12)                 null
);
